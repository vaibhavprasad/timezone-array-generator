var tzArr = [];
moment.tz.names().forEach(item => {
    let temp = {};
    let zone = moment.tz.zone(item);
    temp.id = zone.name;
    temp.offset = offsetMap[`${zone.utcOffset('')}`];
    temp.label = `(${temp.offset}) ${temp.id}`;
    temp.value = zone.name;
    tzArr.push(temp);
})
